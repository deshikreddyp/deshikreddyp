import os
import sys
from dolfin import *
import numpy as np

def angle_between(v1, v2):
    v1 = v1 / np.linalg.norm(v1)
    v2 = v2 / np.linalg.norm(v2)
    angle = np.arccos(abs(np.dot(v1, v2)))
    if np.isnan(angle):
        if (v1 == v2).all():
            return 0.0
        else:
            return 0.0
    return angle

def _mark_loop(bmesh, normals, threshold_angles, boundaries, queue, visited, marker, threshold=30):
    queue_indices = np.where(queue.array())[0]

    num_marked = 0
    emap = bmesh.entity_map(2)
    while len(queue_indices) > 0:
        idx = queue_indices[0]
        facet = Cell(bmesh, idx)
        #FACET = Facet(mesh, emap[idx])
        visited[facet] = True
        queue[facet] = False
        boundaries[facet] = marker
        N = normals[idx]
        threshold_radians = np.pi*threshold_angles[facet]/180.        

	# check if the facets are neighbours and share an edge
	# if they do, then they are in the same boundary
        neighbours = []
        for edge in edges(facet):
            for neighbour in cells(edge):
                if not visited[neighbour]:
                    neighbours.append(neighbour)
	# Also, check if the angle between them is less than threshold
	# get the normal of the cells appended in neighbours
        for neighbour in neighbours:
            n = normals[neighbour.index()]
	# get the angle between the normals
            angle = angle_between(N,n)
            if angle < threshold_radians:
                queue[neighbour] = not visited[neighbour]
        queue_indices = np.where(queue.array())[0]
        num_marked += 1

def mark_boundaries(mesh, threshold_angle=30, I=0, min_facets=10):
    """Mark boundaries automatically by feature edges in mesh. Will connect
    facets to the same boundary if angle between them is less than threshold_angle.
    Will also detect and mark disconnected parts of boundary.
    """
    assert MPI.size(mesh.mpi_comm()) == 1, "Currently only works on non-partitioned meshes."
    bmesh = BoundaryMesh(mesh, "exterior")
    # Number of volumetric domains in the mesh
    visited_facets = MeshFunction("bool", bmesh, bmesh.topology().dim())
    visited_facets.set_all(False)
    bmesh.init(1,2)

    queue = MeshFunction("bool", bmesh, bmesh.topology().dim())
    queue.set_all(False)

    boundaries = MeshFunction("size_t", bmesh,  bmesh.topology().dim())
    boundaries.set_all(0)

    normals = np.zeros((bmesh.num_cells(), 3), dtype=np.float_)
    for facet in cells(bmesh):
        N = Facet(mesh, bmesh.entity_map(2)[facet.index()]).normal()
        normals[facet.index()][0] = N.x()
        normals[facet.index()][1] = N.y()
        normals[facet.index()][2] = N.z()


    threshold_angles = MeshFunction("size_t", bmesh, bmesh.topology().dim())
    threshold_angles.set_all(threshold_angle)

    unvisited = np.where(visited_facets.array() == False)[0]
    # I = 0
    while len(unvisited) > 0:
        I += 1
        queue[unvisited[0]] = True
        _mark_loop(bmesh, normals, threshold_angles, boundaries, queue, visited_facets, I)
        unvisited = np.where(visited_facets.array() == False)[0]        

    # check number of facets in each boundary marker
    # if it is less than 10, then remove it
    # Check for the facet's neighbours and set the boundary marker to the one with the least angle between them
    # This is to avoid having a boundary marker with only one facet
    print(">> Actual number of boundaries detected= "+ str(I))
    print(">> Merging boundaries with less than "+ str(min_facets)+ " facets")

    for i in range(1,I+1):
        facets = np.where(boundaries.array() == i)[0]
        if len(facets) < min_facets:
            for facet in facets:
                neighbours = []
                for edge in edges(Cell(bmesh, facet)):
                    for neighbour in cells(edge):
                        if boundaries[neighbour] != i:
                            neighbours.append(neighbour)
                if len(neighbours) > 0:
                    N = normals[facet]
                    angles = []
                    for neighbour in neighbours:
                        n = normals[neighbour.index()]
                        angles.append(angle_between(N,n))
                    
                boundaries[facet] = boundaries[neighbours[np.argmin(angles)]]

    # renumber the boundaries
    # this is to have the boundary markers in proper order
    print(">> Renumbering boundaries...")
    tag = 0
    for i in range(1,I+1):
        facets = np.where(boundaries.array() == i)[0]
        if len(facets) > 0:
            tag = tag + 1
            boundaries.array()[facets] = tag


    boundaries_full = MeshFunction("size_t", mesh, 2)
    boundaries_full.set_all(0)
    for facet in cells(bmesh):
        boundaries_full[bmesh.entity_map(2)[facet.index()]] = boundaries[facet]
    return boundaries_full, tag
