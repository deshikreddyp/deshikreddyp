mark_boundaries.py

Automated Boundary Tagging

In complex CFD simulations, especially those involving patient-specific geometries, manually tagging boundaries (inlet, outlet, walls) is impractical. The mark_boundaries.py utility automates this by grouping facets based on the angle between their normals.

💡 Algorithm Overview: The script identifies "feature edges" where the angle between neighboring facets exceeds a specified threshold (e.g., 30°). It then uses a region-growing approach to assign unique tags to continuous surface patches.

1. Vector Math utilities

import os
import sys
from dolfin import *
import numpy as np

def angle_between(v1, v2):
    """Compute the angle in radians between two vectors."""
    v1 = v1 / np.linalg.norm(v1)
    v2 = v2 / np.linalg.norm(v2)
    angle = np.arccos(abs(np.clip(np.dot(v1, v2), -1.0, 1.0)))
    return angle

2. Region Growing Loop

def _mark_loop(bmesh, normals, threshold_angles, boundaries, queue, visited, marker):
    queue_indices = np.where(queue.array())[0]
    while len(queue_indices) > 0:
        idx = queue_indices[0]
        facet = Cell(bmesh, idx)
        visited[facet] = True
        queue[facet] = False
        boundaries[facet] = marker
        N = normals[idx]
        threshold_radians = np.pi*threshold_angles[facet]/180.        

        for edge in edges(facet):
            for neighbour in cells(edge):
                if not visited[neighbour]:
                    n = normals[neighbour.index()]
                    if angle_between(N, n) < threshold_radians:
                        queue[neighbour] = True
        queue_indices = np.where(queue.array())[0]

3. Main Marking Function

def mark_boundaries(mesh, threshold_angle=30, I=0, min_facets=10):
    """Mark boundaries automatically by feature edges."""
    bmesh = BoundaryMesh(mesh, "exterior")
    visited_facets = MeshFunction("bool", bmesh, bmesh.topology().dim())
    visited_facets.set_all(False)
    
    boundaries = MeshFunction("size_t", bmesh, bmesh.topology().dim())
    normals = np.zeros((bmesh.num_cells(), 3))
    
    # Store normals for performance
    for facet in cells(bmesh):
        N = Facet(mesh, bmesh.entity_map(2)[facet.index()]).normal()
        normals[facet.index()] = [N.x(), N.y(), N.z()]

    unvisited = np.where(visited_facets.array() == False)[0]
    while len(unvisited) > 0:
        I += 1
        queue = MeshFunction("bool", bmesh, bmesh.topology().dim())
        queue[unvisited[0]] = True
        _mark_loop(bmesh, normals, threshold_angles, boundaries, queue, visited_facets, I)
        unvisited = np.where(visited_facets.array() == False)[0]

    return boundaries, I

⚠️ Cleanup: The full script includes logic to merge small "orphan" facets (less than min_facets) into their nearest neighbors and renumber the tags consecutively.
